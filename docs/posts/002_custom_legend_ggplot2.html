library(ggplot2)
library(grid)
circulos <- circleGrob(
  r = 0.07,
  x = c(0.5, 0.35),
  y = c(0.5,0.5),
  gp = gpar(lwd = 1)
)

lineas <- segmentsGrob(
  x0 = c(0.5,0.35),
  x1 = c(0.5,0.35),
  y0 = c(0.35,0.65),
  y1 = c(0.42,0.58),
  gp =  gpar(lwd = 1)
)

texto <- textGrob(
  label = c("Lower than start","Higher than start"),
  x = c(0.5,0.35),
  y = c(0.3, 0.75),
  gp = gpar(lwd = 2, fontsize = 12),
  )

legend1 <- gTree(children = gList(circulos, lineas,texto))
grid.draw(legend1)
ggplot() +
  guides(
    custom = guide_custom(
    width = unit(5,"cm"),
    height = unit(3,"cm"),
    grob = legend1,
    order = 1
  )) + theme(
    legend.position = "top",
    legend.justification = "center",
    legend.margin = margin(t = -10,b = -20,0,0),
    plot.margin = margin(t = 10,10,10,10)
  )

  x = c(0.02,0.2,0.45,0.70),

)

stick1 <- segmentsGrob(
  x0 =c(0.02,0.2,0.45,0.70),
  x1 = c(0.02,0.2,0.45,0.70),
  y0 = c(0.65,0.42,0.65,0.42),
  y1 = c(0.58,0.35,0.58,0.35),
  gp =  gpar(lwd = 2)
)

label1 <- textGrob(
  label = c("Baby\nBoomers","Gen X", "Milenials", "Gen Z"),
  x = c(0.02,0.2,0.45,0.70),
  y = c(0.8,0.3, 0.75, 0.3),
  gp = gpar(lwd = 2, fontsize = 12, lineheight = 0.7))

circle1 <- circleGrob(
  r = 0.080,
  x = c(0.02,0.2,0.45,0.70),
  y = c(0.5,0.5,0.5,0.5),
  gp = gpar(lty = 1, fill = NA)
)

legend2 <- gTree(children = gList(rect1, rect2, rect3,rect4,
                                 point1,stick1,label1,circle1))
grid.draw(legend2)
ggplot() +
  guides(
    custom = guide_custom(
    width = unit(5,"cm"),
    height = unit(3,"cm"),
    grob = legend2,
    order = 1
  )) + theme(
    legend.position = "top",
    legend.justification = "center",
    legend.margin = margin(t = -10,b = -20,0,0),
    plot.margin = margin(t = 10,10,10,10)
  )
